What is JavaScript?

JavaScript (JS) is the programming language of the web. While HTML provides structure and CSS provides visual styling, JavaScript provides behaviour — making web pages dynamic, interactive, and responsive to user actions in real time. It is the only programming language that runs natively in every major web browser, making it an absolutely essential skill for any web developer.

JavaScript is an event-driven, interpreted language. It interacts with web pages through the DOM (Document Object Model) — a live, programmatic representation of the HTML page structure that JavaScript can read and modify in real time. Using methods like document.getElementById(), document.querySelector(), and properties like .textContent, .innerHTML, and .classList, JavaScript can update content, styles, and structure dynamically without a page reload.

Core JavaScript concepts include: variables (let, const), data types (strings, numbers, arrays, objects, booleans), functions, conditional logic (if/else), loops, and event listeners (addEventListener). ES6+ introduced modern syntax including arrow functions, template literals, destructuring, and powerful array methods such as .map(), .filter(), and .reduce().

In a real-world student dashboard, JavaScript powers all the truly interactive features: dynamically calculating a GPA from entered scores, filtering assignments by deadline, toggling a completed state on tasks, or updating progress bars — all without refreshing the page. JavaScript is what turns a static display into a living, responsive application.

// 2. GRADE CALCULATOR
// ─────────────────────────────────────────────────

// Convert a numeric score into a letter grade
function getLetterGrade(score) {
  if (score === null) return "—";
  if (score >= 70)   return "A";
  if (score >= 60)   return "B";
  if (score >= 50)   return "C";
  if (score >= 40)   return "D";
  return "F";
}

// Calculate average score across graded modules only
function calculateAverage(moduleList) {
  const graded = moduleList.filter(m => m.score !== null);
  if (graded.length === 0) return "N/A";
  const total = graded.reduce((sum, m) => sum + m.score, 0);
  return (total / graded.length).toFixed(1);
}

// Build grades table dynamically in the DOM
function renderGradesTable() {
  const tbody = document.getElementById("gradesTableBody");
  tbody.innerHTML = ""; // Clear before re-render

  modules.forEach(mod => {
    const grade = getLetterGrade(mod.score);
    const row   = document.createElement("tr");

    row.innerHTML = `
      ${mod.name}
      ${mod.score !== null ? mod.score + "%" : "Pending"}
      ${grade}
    `;
    tbody.appendChild(row);
  });

  // Update average display
  const avg   = calculateAverage(modules);
  const avgEl = document.getElementById("averageDisplay");
  avgEl.textContent = avg !== "N/A" ? avg + "%" : avg;
  avgEl.style.color = parseFloat(avg) >= 70
    ? "#28a745"
    : parseFloat(avg) >= 50 ? "#ffc107" : "#dc3545";
}

// ─────────────────────────────────────────────────
// 3. SCORE INPUT — Add a score for CET138 live
// ─────────────────────────────────────────────────
function handleScoreInput() {
  const input    = document.getElementById("scoreInput");
  const feedback = document.getElementById("scoreFeedback");
  const score    = parseFloat(input.value);

  if (isNaN(score) || score < 0 || score > 100) {
    feedback.textContent = "⚠️ Please enter a value between 0 and 100.";
    feedback.style.color = "#dc3545";
    return;
  }

  modules[0].score = score;
  renderGradesTable();

  const grade = getLetterGrade(score);
  feedback.textContent =
    `✅ Score saved! Grade: ${grade}. New average: ${calculateAverage(modules)}%`;
  feedback.style.color = "#28a745";
  input.value = "";
}

// ─────────────────────────────────────────────────
// 4. ASSIGNMENT MANAGER — Add & toggle tasks
// ─────────────────────────────────────────────────
const assignments = [
  { title: "CET138 ePortfolio",  due: "30 March 2025", done: false },
  { title: "CET120 ER Diagram",  due: "5 April 2025",  done: false }
];

function renderAssignments() {
  const list = document.getElementById("assignmentList");
  list.innerHTML = "";

  assignments.forEach((task, index) => {
    const li = document.createElement("li");
    li.style.cssText =
      `padding: 0.75rem 1rem;
       display: flex; justify-content: space-between;
       border-bottom: 1px solid #eee;
       text-decoration: ${task.done ? "line-through" : "none"};
       opacity: ${task.done ? "0.5" : "1"};`;

    li.innerHTML = `
      ${task.done ? "✅" : "⏰"} 
        ${task.title} – Due: ${task.due}
      
      
    `;
    list.appendChild(li);
  });

  const pending = assignments.filter(a => !a.done).length;
  document.getElementById("pendingCount").textContent = pending;
}

// Toggle a task's completion state
function toggleDone(index) {
  assignments[index].done = !assignments[index].done;
  renderAssignments();
}

// Add a new assignment dynamically
function addAssignment() {
  const titleEl = document.getElementById("newTaskTitle");
  const dueEl   = document.getElementById("newTaskDue");

  if (!titleEl.value.trim() || !dueEl.value) {
    alert("Please fill in both the task title and due date.");
    return;
  }

  assignments.push({ title: titleEl.value.trim(), due: dueEl.value, done: false });
  titleEl.value = "";
  dueEl.value   = "";
  renderAssignments();
}

// ─────────────────────────────────────────────────
// 5. INITIALISE on DOM ready
// ─────────────────────────────────────────────────
document.addEventListener("DOMContentLoaded", () => {
  renderGradesTable();
  renderAssignments();
  console.log("✅ Dashboard initialised. Modules:", modules.length);
});
